Clave
Resolvers = Adaptadores
Los resolvers solo llaman a los casos de uso existentes.
// Los resolvers NO tienen lógica de negocio
export const createResolvers = (
  createTask: CreateTask,
  getAllTasks: GetAllTasks
) => ({
  Query: {
    tasks: () => getAllTasks.execute()
  },
  Mutation: {
    createTask: (_, { input }) =>
      createTask.execute(input.title)
  }
});

Magia
Mismos casos de uso, dos adaptadores
// index.ts - Punto de entrada

// 1. Crear instancias compartidas
const taskRepository = new InMemoryTaskRepository();
const createTask = new CreateTask(taskRepository);
const getAllTasks = new GetAllTasks(taskRepository);

// 2. Conectar AMBOS adaptadores
const restAdapter = new ExpressAdapter(createTask, getAllTasks);
const graphqlAdapter = new ApolloAdapter(createTask, getAllTasks);

// 3. Ambos funcionan en paralelo!
restAdapter.setupRoutes(app);    // /tasks
graphqlAdapter.applyMiddleware(app); // /graphql
